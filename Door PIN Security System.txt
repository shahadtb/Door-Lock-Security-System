  pinMode(RLed, OUTPUT);  pinMode(GLed, OUTPUT);  alarmsetup(true); //boolean statement for default conditions}void loop(){ if (Serial.available() > 0){ //if serial input is availble    userinput = Serial.readStringUntil('\n');//read the user input        Serial.println("\n"); // new line   if(userinput == passcode){ //if user input is equal to the original passcode          alarmsetup(false); // run the alarmsetup function, while the boolean is false, running the false part         }   if(userinput != passcode){ //if the user input is incorrect     Serial.println("passcode is inncorrect!"); //asl the user to enter the correct passcode   }   alarmsetup(true); //run default condition from the alarmsetup function }}//this fucntion includes the default setting for the led lights and buzzer depeding on the user's entry //this function also includes the set up changes if the the entered passcode is correctvoid alarmsetup(bool alarmdf){ //alarm default setting boolean conditions    if (alarmdf== true){ // when the condition is true, run default setting of the lights and buzzer  digitalWrite(RLed, HIGH); //default red light is on  digitalWrite(GLed, LOW);//default red light is off  tone(buzzer,1000);// default buzzer tone from requirments  delay(500); //delay to allow the buzzer to run for 1/2 second  noTone(buzzer); //stop the tone of the buzzer  Serial.println("Please enter 4-digit pin");//asking the user to type the passcode}  if (alarmdf== false){ //when the passcode entry is correct   Serial.println("Welcome...The passcode entered is:"); //print welcome statement  Serial.println(userinput); //print the user entry  digitalWrite(RLed, LOW); //turn off red led  digitalWrite(GLed, HIGH);//turn on green led  tone(buzzer,1000);  delay(5000);//buzzer sounds for 5 seconds which is 5000ms  noTone(buzzer);//stop the tone of buzzer   delay(800); //delay to allow the repeat of the code and tone to be clear}}
